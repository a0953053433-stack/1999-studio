<script>
const addons = [
{ id: 'sa', name: '簡單手繪', price: 50 },
{ id: 'ma', name: '中等手繪', price: 150 },
{ id: 'aa', name: '高級手繪', price: 250 },
{ id: 'an', name: '動漫正比', price: 500 },
{ id: 'fr', name: '法式', price: 50 },
{ id: 'gr', name: '漸層', price: 20 },
{ id: 'pw', name: '蹭粉', price: 40 },
{ id: 'sd', name: '小鑽', price: 10 },
{ id: 'md', name: '中鑽', price: 20 },
{ id: 'ld', name: '大鑽', price: 50 }
];
let state = {
base: { id: 'solid', name: '單色基礎', price: 800 },
rem: { id: 'none', name: '不需卸甲', price: 0 },
qty: {}
};
addons.forEach(a => state.qty[a.id] = 0);
function renderAddons() {
document.getElementById('addon-list').innerHTML = addons.map(a => `
<div class="flex items-center justify-between p-5 bg-zinc-900/80 rounded-3xl border border-zinc-800 shadow-inner">
<div>
<div class="font-black text-white text-lg">${a.name}</div>
<div class="text-[10px] text-zinc-500 font-bold tracking-[0.2em]">$${a.price} / 指</div>
</div>
<div class="flex items-center gap-6">
<button onclick="updateQty('${a.id}', -1)" class="w-10 h-10 flex items-center justify-center rounded-2xl bg-zinc-800 text-pink-500 border border-zinc-700 active:bg-zinc-700 transition-all font-black">-</button>
<span class="w-6 text-center font-black text-3xl num-font ${state.qty[a.id] > 0 ? 'text-white' : 'text-zinc-800'}">${state.qty[a.id]}</span>
<button onclick="updateQty('${a.id}', 1)" class="w-10 h-10 flex items-center justify-center rounded-2xl bg-pink-600 text-white shadow-xl active:bg-pink-500 transition-all font-black">+</button>
</div>
</div>
`).join('');
calc();
}
function setBase(id, price, name) {
state.base = { id, price, name };
document.querySelectorAll('[id^="base-"]').forEach(el => el.classList.remove('active-ring'));
document.getElementById('base-' + id).classList.add('active-ring');
calc();
}
function setRemoval(id, price, name) {
state.rem = { id, price, name };
document.querySelectorAll('.rem-btn').forEach(el => el.classList.remove('active-ring'));
document.getElementById('rem-' + id).classList.add('active-ring');
calc();
}
function updateQty(id, delta) {
state.qty[id] = Math.max(0, state.qty[id] + delta);
renderAddons();
}
function calc() {
let addTotal = 0;
addons.forEach(a => addTotal += a.price * state.qty[a.id]);
document.getElementById('total-price').innerText = state.base.price + addTotal + state.rem.price;
}
function resetApp() {
addons.forEach(a => state.qty[a.id] = 0);
setBase('solid', 800, '單色基礎');
setRemoval('none', 0, '不需卸甲');
renderAddons();
}
function copyQuote() {
let t = `💖 1999 studio 報價單 💖\n------------------\n✨ 基礎：${state.base.name} $${state.base.price}\n`;
addons.forEach(a => {
if(state.qty[a.id] > 0) t += `💎 ${a.name} (${state.qty[a.id]}指) +$${a.price * state.qty[a.id]}\n`;
});
if(state.rem.id !== 'none') t += `🧼 卸甲：${state.rem.name} +$${state.rem.price}\n`;
t += `------------------\n🎀 總計金額：$${document.getElementById('total-price').innerText}\n期待見到妳唷！Chu~ 😘`;
try {
const el = document.createElement('textarea');
el.value = t;
document.body.appendChild(el);
el.select();
document.execCommand('copy');
document.body.removeChild(el);
const btn = document.getElementById('copy-btn');
const text = document.getElementById('copy-text');
btn.style.backgroundColor = '#22c55e';
text.innerText = "DONE!";
text.style.color = "white";
setTimeout(() => {
btn.style.backgroundColor = 'white';
text.innerText = "COPY";
text.style.color = "black";
}, 1500);
} catch (err) { alert('複製失敗'); }
}
resetApp();
</script>
